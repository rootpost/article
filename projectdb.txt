CREATE TABLE article(
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  title TEXT,
  category INTEGER,
  description TEXT,
  source TEXT,
  datetime INTEGER,
  tagname INTEGER
)

CREATE TABLE category(
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT
)

CREATE TABLE tags(
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT
)
